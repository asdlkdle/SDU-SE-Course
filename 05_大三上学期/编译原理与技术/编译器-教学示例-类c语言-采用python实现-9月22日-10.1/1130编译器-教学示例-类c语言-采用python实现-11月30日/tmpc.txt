int main(){
    int array[4] = {7,3,5,9};
    int len;
    len = 4;
    int i;
    i=1;
    while(i<=len-1){
        int min,j;
        min = i;
        j=i;
        while(j<=len){
            if(array[min]>array[j])then{
                min = j;
            }
            j = j+1;
        }
        int temp;
        temp = array[min];
        array[min] = array[i];
        array[i] = temp;
        i = i+1;
    }

	return array[1];
}
